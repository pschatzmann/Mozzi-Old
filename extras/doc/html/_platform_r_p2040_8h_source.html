140  void setupPWM(const int16_t pins[]) {
141  // setup pwm frequency
142  pwm_config pico_pwm_config = pwm_get_default_config();
143  int wrap_value = AUDIO_BIAS*2; // amplitude of square wave (pwm values -amplitude to amplitude) for one byte
144  float pwmClockDivider = static_cast<float>(clock_get_hz(clk_sys)) / (PWM_RATE * wrap_value);
145  pwm_config_set_clkdiv(&pico_pwm_config, pwmClockDivider);
146  pwm_config_set_clkdiv_mode(&pico_pwm_config, PWM_DIV_FREE_RUNNING);
147  pwm_config_set_wrap (&pico_pwm_config, wrap_value);
148 
149  // setup pins
150  for (int j=0;j<AUDIO_CHANNELS;j++){
151  int16_t gpio = pins[j];
152  if (gpio >- 1) {
153  sprintf(msg,"Channel %d -> pin %d", j, gpio);
154  Serial.println(msg);
155  setupPWMPin(gpio, pico_pwm_config);
156  max_channel_pico = j + 1;
157  } else {
158  sprintf(msg,"Channel %d -> pin %d is already defined", j, gpio);
159  Serial.println(msg);
160  }
161  }
162  }
163 
164  void stopPWM(const int16_t pins[]){
165  // stop all pins
166  for (int j=0;j<max_channel_pico;j++){
167  int gpio = pins[j];
168  if (gpio >- 1) {
169  uint slice = pwm_gpio_to_slice_num(gpio);
170  pwm_set_enabled(slice, false);
171  }
172  }
173  }
