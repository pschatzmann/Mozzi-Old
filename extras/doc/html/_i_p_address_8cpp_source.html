49 bool IPAddress::fromString(const char *address)
50 {
51  // TODO: add support for "a", "a.b", "a.b.c" formats
52 
53  int16_t acc = -1; // Accumulator
54  uint8_t dots = 0;
55 
56  while (*address)
57  {
58  char c = *address++;
59  if (c >= '0' && c <= '9')
60  {
61  acc = (acc < 0) ? (c - '0') : acc * 10 + (c - '0');
62  if (acc > 255) {
63  // Value out of [0..255] range
64  return false;
65  }
66  }
67  else if (c == '.')
68  {
69  if (dots == 3) {
70  // Too much dots (there must be 3 dots)
71  return false;
72  }
73  if (acc < 0) {
74  /* No value between dots, e.g. '1..' */
75  return false;
76  }
77  _address.bytes[dots++] = acc;
78  acc = -1;
79  }
80  else
81  {
82  // Invalid char
83  return false;
84  }
85  }
86 
87  if (dots != 3) {
88  // Too few dots (there must be 3 dots)
89  return false;
90  }
91  if (acc < 0) {
92  /* No value between dots, e.g. '1..' */
93  return false;
94  }
95  _address.bytes[3] = acc;
96  return true;
97 }
