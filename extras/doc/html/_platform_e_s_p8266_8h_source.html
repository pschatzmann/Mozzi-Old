173  static void startAudioStandard() {
174 
175  #if IS_ESP8266 && (EXTERNAL_AUDIO_OUTPUT == true) && (BYPASS_MOZZI_OUTPUT_BUFFER != true) // for external audio output, set up a timer running a audio rate
176  timer1_isr_init();
177  timer1_attachInterrupt(defaultAudioOutput);
178  timer1_enable(TIM_DIV1, TIM_EDGE, TIM_LOOP);
179  timer1_write(F_CPU / audio_rate);
180  #else
181  #if (ESP_AUDIO_OUT_MODE == PDM_VIA_SERIAL)
182  Serial1.begin(audio_rate * (PDM_RESOLUTION * 40), SERIAL_8N1,
183  SERIAL_TX_ONLY); // Note: PDM_RESOLUTION corresponds to packets of 32
184  // encoded bits However, the UART (unfortunately) adds a
185  // start and stop bit each around each byte, thus sending
186  // a total to 40 bits per audio sample per
187  // PDM_RESOLUTION.
188  // set up a timer to copy from Mozzi output_buffer into Serial TX buffer
189  timer1_isr_init();
190  timer1_attachInterrupt(esp8266_serial_audio_output);
191  // UART FIFO buffer size is 128 bytes. To be on the safe side, we keep the
192  // interval to the time needed to write half of that. PDM_RESOLUTION * 4 bytes
193  // per sample written.
194  timer1_enable(TIM_DIV16, TIM_EDGE, TIM_LOOP);
195  timer1_write(F_CPU / (audio_rate * PDM_RESOLUTION));
196  #else
197  i2s_begin();
198  #if (ESP_AUDIO_OUT_MODE == PDM_VIA_I2S)
199  pinMode(2, INPUT); // Set the two unneeded I2S pins to input mode, to reduce
200  // side effects
201  pinMode(15, INPUT);
202  #endif
203  i2s_set_rate(audio_rate * PDM_RESOLUTION);
204  if (output_buffer_size == 0)
205  output_buffer_size =
206  i2s_available(); // Do not reset count when stopping / restarting
207  #endif
208  #endif
209  }
210 
211  #endif

221 void CACHED_FUNCTION_ATTR esp8266_serial_audio_output() {
222  // Note: That unreadble mess is an optimized version of Serial1.availableForWrite()
223  while ((UART_TX_FIFO_SIZE - ((U1S >> USTXC) & 0xff)) > (PDM_RESOLUTION * 4)) {
