144  void setupPWM(const int16_t pins[]) {
145  for (int j=0;j<AUDIO_CHANNELS;j++){
146  int16_t gpio = pins[j];
147  if (pwm_pins[j] == nullptr && gpio >- 1) {
148  mbed::PwmOut *pin = new mbed::PwmOut(digitalPinToPinName(gpio));
149  pwm_pins[j] = pin;
150  if (pin!=nullptr) {
151  sprintf(msg,"Channel %d -> pin %d", j, gpio);
152  Serial.println(msg);
153  setupPWMPin(*pin);
154  max_channel = j + 1;
155  } else {
156  sprintf(msg,"Channel %d -> pin could not be defined", j);
157  Serial.println(msg);
158  }
159  } else {
160  sprintf(msg,"Channel %d -> pin %d is already defined", j, gpio);
161  Serial.println(msg);
162  }
163  }
164  }
