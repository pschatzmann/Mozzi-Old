81 void stm32_adc_eoc_handler()
82 {
83  switch (adc_count) {
84  case 0:
85  // 6us
86  receiveSecondAudioADC();
87  adcReadSelectedChannels();
88  break;
89 
90  case 1:
91  // <2us, <1us w/o receive
92  // receiveFirstControlADC();
93  startSecondControlADC();
94  break;
95 
96  case 2:
97  // 3us
98  receiveSecondControlADC();
99  startFirstAudioADC();
100  break;
101  }
102  adc_count++;
103 }

137 static void startAudioStandard() {
138 
139  audio_update_timer.pause();
140  //audio_update_timer.setPeriod(1000000UL / AUDIO_RATE);
141  // Manually calculate prescaler and overflow instead of using setPeriod, to avoid rounding errors
142  uint32_t period_cyc = F_CPU / audio_rate;
143  uint16_t prescaler = (uint16_t)(period_cyc / 65535 + 1);
144  uint16_t overflow = (uint16_t)((period_cyc + (prescaler / 2)) / prescaler);
145  audio_update_timer.setPrescaleFactor(prescaler);
146  audio_update_timer.setOverflow(overflow);
147  audio_update_timer.setChannel1Mode(TIMER_OUTPUT_COMPARE);
148  audio_update_timer.setCompare(TIMER_CH1,
149  1); // Interrupt 1 count after each update
150  audio_update_timer.attachCompare1Interrupt(defaultAudioOutput);
151  audio_update_timer.refresh();
152  audio_update_timer.resume();
153 
154 #if (EXTERNAL_AUDIO_OUTPUT != true)
155  pinMode(AUDIO_CHANNEL_1_PIN, PWM);
156  #if (AUDIO_MODE == HIFI)
157  pinMode(AUDIO_CHANNEL_1_PIN_HIGH, PWM);
158  #elif (AUDIO_CHANNELS > 1)
159  pinMode(AUDIO_CHANNEL_2_PIN, PWM);
160  #endif
161 
162  #define MAX_CARRIER_FREQ (F_CPU / (1 << AUDIO_BITS_PER_CHANNEL))
163  #if MAX_CARRIER_FREQ < AUDIO_RATE
164  #error Configured audio resolution is definitely too high at the configured audio rate (and the given CPU speed)
165  #elif MAX_CARRIER_FREQ < (AUDIO_RATE * 3)
166  #warning Configured audio resolution may be higher than optimal at the configured audio rate (and the given CPU speed)
167  #endif
168 
169  if (MAX_CARRIER_FREQ < (audio_rate * 5)) {
170  // Generate as fast a carrier as possible
171  audio_pwm_timer.setPrescaleFactor(1);
172  } else {
173  // No point in generating arbitrarily high carrier frequencies. In fact, if
174  // there _is_ any headroom, give the PWM pin more time to swing from HIGH to
175  // LOW and BACK, cleanly
176  audio_pwm_timer.setPrescaleFactor((int)MAX_CARRIER_FREQ / (audio_rate * 5));
177  }
178  audio_pwm_timer.setOverflow(1 << AUDIO_BITS_PER_CHANNEL); // Allocate enough room to write all
179  // intended bits
180 #endif
181 
182 }

192 void PlatformSTM32::start(int control_rate_hz) {
193  setupMozziADC(); // you can use setupFastAnalogRead() with FASTER or FASTEST
194  // in setup() if desired (not for Teensy 3.* )
195  // delay(200); // so AutoRange doesn't read 0 to start with
196  startControl(control_rate_hz);
197 #if (AUDIO_MODE == STANDARD) || (AUDIO_MODE == STANDARD_PLUS) || \
198  IS_STM32 // Sorry, this is really hacky. But on STM32 regular and HIFI
199  // audio modes are so similar to set up, that we do it all in one
200  // function.
201  startAudioStandard();
202 #elif (AUDIO_MODE == HIFI)
203  startAudioHiFi();
204 #endif
205 }
