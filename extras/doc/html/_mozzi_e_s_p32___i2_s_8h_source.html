84  static void startAudioStandard() {
85  static const i2s_config_t i2s_config = {
86  .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_TX),
87  .sample_rate = AUDIO_RATE * PDM_RESOLUTION,
88  .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT, // only the top 8 bits will actually be used by the internal DAC, but using 8 bits straight away seems buggy
89  .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT, // always use stereo output. mono seems to be buggy, and the overhead is insignifcant on the ESP32
90  .communication_format = (i2s_comm_format_t)(I2S_COMM_FORMAT_I2S | I2S_COMM_FORMAT_I2S_LSB), // this appears to be the correct setting for internal DAC and PT8211, but not for other dacs
91  .intr_alloc_flags = 0, // default interrupt priority
92  .dma_buf_count = 8, // 8*128 bytes of buffer corresponds to 256 samples (2 channels, see above, 2 bytes per sample per channel)
93  .dma_buf_len = 128,
94  .use_apll = false
95  };

156  #if (EXTERNAL_AUDIO_OUTPUT != true)
157  // On ESP32 we cannot test wether the DMA buffer has room. Instead, we have to use a one-sample mini buffer. In each iteration we
158  // _try_ to write that sample to the DMA buffer, and if successful, we can buffer the next sample. Somewhat cumbersome, but works.
159  // TODO: Should ESP32 gain an implemenation of i2s_available(), we should switch to using that, instead.
160  bool _esp32_can_buffer_next = true;
161  #if (ESP32_AUDIO_OUT_MODE == INTERNAL_DAC)
162  uint16_t _esp32_prev_sample[2];
163  #define ESP_SAMPLE_SIZE (2*sizeof(uint16_t))
164  #elif (ESP32_AUDIO_OUT_MODE == PT8211_DAC)
165  int16_t _esp32_prev_sample[2];
166  #define ESP_SAMPLE_SIZE (2*sizeof(int16_t))
167  #elif (ESP32_AUDIO_OUT_MODE == PDM_VIA_I2S)
168  uint32_t _esp32_prev_sample[PDM_RESOLUTION];
169  #define ESP_SAMPLE_SIZE (PDM_RESOLUTION*sizeof(uint32_t))
170  #endif
171 
172  inline bool esp32_tryWriteSample() {
173  size_t bytes_written;
174  i2s_write(i2s_num, &_esp32_prev_sample, ESP_SAMPLE_SIZE, &bytes_written, 0);
175  return (bytes_written != 0);
176  }
177 
178  inline bool canBufferAudioOutput() {
179  if (_esp32_can_buffer_next) return true;
180  _esp32_can_buffer_next = esp32_tryWriteSample();
181  return _esp32_can_buffer_next;
182  }
183 
184  inline void audioOutput(const AudioOutput f) {
185  #if (ESP32_AUDIO_OUT_MODE == INTERNAL_DAC)
186  _esp32_prev_sample[0] = (f.l() + AUDIO_BIAS) << 8;
187  #if (AUDIO_CHANNELS > 1)
188  _esp32_prev_sample[1] = (f.r() + AUDIO_BIAS) << 8;
189  #else
190  // For simplicity of code, even in mono, we're writing stereo samples
191  _esp32_prev_sample[1] = _esp32_prev_sample[0];
192  #endif
193  #elif (ESP32_AUDIO_OUT_MODE == PDM_VIA_I2S)
194  for (uint8_t i=0; i<PDM_RESOLUTION; ++i) {
195  _esp32_prev_sample[i] = pdmCode32(f.l() + AUDIO_BIAS);
196  }
197  #else
198  // PT8211 takes signed samples
199  _esp32_prev_sample[0] = f.l();
200  _esp32_prev_sample[1] = f.r();
201  #endif
202  _esp32_can_buffer_next = esp32_tryWriteSample();
203  }
204  #else
205  inline bool canBufferAudioOutput() {return true;}
206 
207  #endif
208 
209  inline void bufferAudioOutput(const AudioOutput_t f) {
210  audioOutput(f);
211  ++samples_written_to_buffer;
212  }
