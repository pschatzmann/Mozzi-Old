223 inline uint32_t pdmCode8(uint16_t sample) {
224  // lookup table for fast pdm coding on 8 output bits at a time
225  static const byte fast_pdm_table[]{0, 0b00010000, 0b01000100,
226  0b10010010, 0b10101010, 0b10110101,
227  0b11011101, 0b11110111, 0b11111111};
228 
229  static uint32_t lastwritten = 0;
230  static uint32_t nexttarget = 0;
231  // in each iteration, code the highest 3-and-a-little bits.
232  // Note that sample only has 16 bits, while the
233  // highest bit we consider for writing is bit 17.
234  // Thus, if the highest bit is set, the next
235  // three bits cannot be.
236  nexttarget += sample;
237  nexttarget -= lastwritten;
238  lastwritten = nexttarget & 0b11110000000000000;
239  return fast_pdm_table[lastwritten >> 13];
240 }

243 inline uint32_t pdmCode32(uint16_t sample) {
244  uint32_t outbits = 0;
245  for (uint8_t i = 0; i < 4; ++i) {
246  outbits = outbits << 8;
247  outbits |= pdmCode8(sample);
248  }
249  return outbits;
250 }
